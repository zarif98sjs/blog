    for(int i=1;i<=tc;i++)
    {
        int x,y;
        char ch;
        cin>>x>>ch>>y;

        double a,b=(y/(x*1.0))*a;

        double target=200; /// 400/2

        double lo=0,hi=200,mid=(lo+hi)/2;
        a=mid;
        b=(y/(x*1.0))*a;

        double val = sqrt(a*a+b*b)*atan(b/(a*1.0)) + a;

        while(1)
        {
            if(fabs(val-target)<=EPS)
                break;
            else if(val>target)
                hi=mid;
            else if(val<target)
                lo=mid;

            mid = (lo+hi)/2;
            a = mid;
            b = (y/(x*1.0))*a;

            val = sqrt(a*a+b*b)*atan(b/(a*1.0)) + a;
        }

    }

}
