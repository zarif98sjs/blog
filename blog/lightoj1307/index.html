  • For a triangle , the sum of any 2 side should be greater than the third side , which is to say the sum of the 2 shorter sides should be greater than the third side.
    i.e a + b > c [where a<=b<=c]

int ara[nmax];

int main()
{
    //freopen("out.txt","w",stdout);

    optimizeIO();

    int tc;
    cin>>tc;

    for(int q=1;q<=tc;q++)
    {
        int n;
        cin>>n;

        for(int i=1;i<=n;i++)
            cin>>ara[i];

        sort(ara+1,ara+n+1);

        LL ans = 0;

        for(int i=1;i<=n;i++)
        {
            for(int j=i+1;j<=n;j++)
            {
                int s = ara[i] + ara[j];
                int val = lower_bound(ara+1,ara+n+1,s) - ara -1;

                val -= j;
                ans = ans + val;

            }
        }

        cout<<"Case "<<q<<": ";
        cout<<ans<<endl;

    }

    return 0;
}
